## C++

```C++
#include<iostream>
using namespace std;
int main() {
    int m, k;
    while (cin >> m >> k) {
        if (m == 0 && k == 0) break;
        int sum = m + m / k; // 第一轮回得到总话费
        int count = m / k + m % k; // 额外话费 = 赠送+剩余话费
        while (count / k != 0) { // 只要额外话费大于 k 就还能获得 剩余话费
            sum += count / k;
            count = count / k + count % k;
        }
        cout << sum << endl;
    }
}


```

package main

import (
	"fmt"
)

func main() {
	var m, k int
	for {
		_, err := fmt.Scanf("%d %d", &m, &k)
		if err != nil || (m == 0 && k == 0) {
			break
		}

		sum := m + m/k
		count := m/k + m%k

		for count/k != 0 {
			sum += count / k
			count = count/k + count%k
		}

		fmt.Println(sum)
	}
}

const rl = require('readline').createInterface({
    input:process.stdin,
    output:process.stdout
})

function Day() {
    rl.on('line',(input)=>{
        const part = input.split(' ').map(Number)
        let M = part[0], K = part[1]
        let unuse = M  // 未参与奖励规则的话费
        let day = M    // 充了多少话费目前就能用多少天
        if(M == 0 && K == 0){
            return
        }   
        while( Math.floor(unuse / K) !== 0){ // 当未参与奖励的还大于 k 就能继续获得奖励
            let free = Math.floor(unuse / K)
            if(unuse % K == 0){
                unuse = 0
            }else{
               unuse = free + (unuse % K)
            }
            day += free
        }
        console.log(day)
    })
}
